// functions/admin-actions.js // POST /api/admin-actions // Body: { action: "block_time"|"cancel"|"reschedule"|"manual_book", ...params } // All admin actions in one handler, dispatched by action field const { getSupabase } = require('./utils/supabase'); const { jsonResponse, corsHeaders, isUUID } = require('./utils/validate'); const { requireAdmin } = require('./utils/auth'); const { validateSlotAvailable, computeTotalDuration, naiveDatetimeToUtc } = require('./utils/availability'); exports.handler = async (event) => { if (event.httpMethod === 'OPTIONS') return corsHeaders(); if (event.httpMethod !== 'POST') return jsonResponse(405, { error: 'Method not allowed' }); let body; try { body = JSON.parse(event.body); } catch { return jsonResponse(400, { error: 'Invalid request' }); } try { const supabase = getSupabase(); const { businessId } = await requireAdmin(event, supabase); switch (body.action) { case 'block_time': return await blockTime(supabase, businessId, body); case 'cancel': return await cancelBooking(supabase, businessId, body); case 'reschedule': return await rescheduleBooking(supabase, businessId, body); case 'manual_book': return await manualBook(supabase, businessId, body); default: return jsonResponse(400, { error: 'Unknown action' }); } } catch (err) { if (err.status) return jsonResponse(err.status, { error: err.message }); console.error('Admin action error:', err); return jsonResponse(500, { error: 'Action failed' }); } }; // Block time (one-off) async function blockTime(supabase, businessId, body) { const { name, start_at, end_at } = body; if (!start_at || !end_at) { return jsonResponse(400, { error: 'Start and end times are required' }); } // Look up business timezone to convert local times to UTC const { data: business } = await supabase .from('businesses') .select('timezone') .eq('id', businessId) .single(); const tz = business?.timezone || 'America/New_York'; // Parse naive datetime strings (e.g. "2026-03-04T12:00:00") as local business time const startUtc = naiveDatetimeToUtc(start_at, tz); const endUtc = naiveDatetimeToUtc(end_at, tz); if (!startUtc || !endUtc || endUtc <= startUtc) { return jsonResponse(400, { error: 'Invalid time range' }); } const { data, error } = await supabase .from('one_off_blocks') .insert({ business_id: businessId, name: name || 'Blocked', start_at: startUtc.toISOString(), end_at: endUtc.toISOString() }) .select() .single(); if (error) { console.error('Block time error:', error); return jsonResponse(500, { error: 'Failed to block time' }); } return jsonResponse(201, { block: data }); } // Cancel booking async function cancelBooking(supabase, businessId, body) { const { booking_id, internal_notes } = body; if (!booking_id || !isUUID(booking_id)) { return jsonResponse(400, { error: 'Invalid booking ID' }); } const { data, error } = await supabase .from('bookings') .update({ status: 'cancelled', internal_notes: internal_notes || null }) .eq('id', booking_id) .eq('business_id', businessId) .eq('status', 'confirmed') .select() .single(); if (error || !data) { return jsonResponse(404, { error: 'Booking not found or already cancelled' }); } return jsonResponse(200, { booking: data }); } // Reschedule booking async function rescheduleBooking(supabase, businessId, body) { const { booking_id, new_start_at } = body; if (!booking_id || !isUUID(booking_id) || !new_start_at) { return jsonResponse(400, { error: 'Booking ID and new time are required' }); } // Fetch existing booking const { data: existing } = await supabase .from('bookings') .select('*, booking_items(*)') .eq('id', booking_id) .eq('business_id', businessId) .eq('status', 'confirmed') .single(); if (!existing) { return jsonResponse(404, { error: 'Booking not found' }); } // Calculate duration from existing booking const origDuration = (new Date(existing.end_at) - new Date(existing.start_at)) / 60000; // Fetch business data for validation const { data: business } = await supabase .from('businesses') .select('timezone') .eq('id', businessId) .single(); const tz = business?.timezone || 'America/New_York'; // Convert naive local time to UTC const newStart = naiveDatetimeToUtc(new_start_at, tz); if (!newStart) return jsonResponse(400, { error: 'Invalid time format' }); const newEnd = new Date(newStart.getTime() + origDuration * 60000); const [settingsRes, rulesRes, blackoutsRes] = await Promise.all([ supabase.from('settings').select('*').eq('business_id', businessId).single(), supabase.from('availability_rules').select('*').eq('business_id', businessId), supabase.from('blackout_blocks').select('*').eq('business_id', businessId).eq('is_active', true) ]); // Temporarily cancel old booking for collision check // (We don't want it to collide with itself) await supabase .from('bookings') .update({ status: 'cancelled' }) .eq('id', booking_id); try { const validation = await validateSlotAvailable(supabase, { businessId, timezone: business.timezone, startAt: newStart.toISOString(), totalDurationMinutes: origDuration, settings: settingsRes.data || {}, availabilityRules: rulesRes.data || [], blackoutBlocks: blackoutsRes.data || [] }); if (!validation.valid) { // Restore original booking await supabase .from('bookings') .update({ status: 'confirmed' }) .eq('id', booking_id); return jsonResponse(409, { error: 'The new time is not available' }); } // Update booking with new time const { data: updated, error: updateErr } = await supabase .from('bookings') .update({ start_at: newStart.toISOString(), end_at: newEnd.toISOString(), status: 'confirmed' }) .eq('id', booking_id) .select() .single(); if (updateErr) { await supabase.from('bookings').update({ status: 'confirmed' }).eq('id', booking_id); return jsonResponse(500, { error: 'Failed to reschedule' }); } return jsonResponse(200, { booking: updated }); } catch (err) { // Restore on any error await supabase.from('bookings').update({ status: 'confirmed' }).eq('id', booking_id); throw err; } } // Manual booking (walk-in or phone booking) async function manualBook(supabase, businessId, body) { const { start_at, end_at, customer_name, customer_phone, customer_email, is_new_client, internal_notes, services_text } = body; if (!start_at || !end_at || !customer_name) { return jsonResponse(400, { error: 'Start time, end time, and customer name are required' }); } // Look up business timezone const { data: business } = await supabase .from('businesses') .select('timezone') .eq('id', businessId) .single(); const tz = business?.timezone || 'America/New_York'; const startUtc = naiveDatetimeToUtc(start_at, tz); const endUtc = naiveDatetimeToUtc(end_at, tz); const { data: booking, error } = await supabase .from('bookings') .insert({ business_id: businessId, start_at: startUtc.toISOString(), end_at: endUtc.toISOString(), customer_name: customer_name.trim(), customer_phone: customer_phone?.trim() || null, customer_email: customer_email?.trim() || null, is_new_client: !!is_new_client, status: 'confirmed', internal_notes: internal_notes || services_text || null }) .select() .single(); if (error) { console.error('Manual book error:', error); return jsonResponse(500, { error: 'Failed to create booking' }); } return jsonResponse(201, { booking }); }
