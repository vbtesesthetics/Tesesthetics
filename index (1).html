// functions/admin-export.js // GET /api/admin-export?type=bookings|clients|revenue&date_from=&date_to= // Returns CSV data for download const { getSupabase } = require('./utils/supabase'); const { requireAdmin } = require('./utils/auth'); exports.handler = async (event) => { if (event.httpMethod === 'OPTIONS') return cors(); if (event.httpMethod !== 'GET') return res(405, 'Method not allowed'); try { const supabase = getSupabase(); const { businessId } = await requireAdmin(event, supabase); const type = event.queryStringParameters?.type || 'bookings'; const dateFrom = event.queryStringParameters?.date_from; const dateTo = event.queryStringParameters?.date_to; let csv = ''; let filename = ''; switch (type) { case 'bookings': csv = await exportBookings(supabase, businessId, dateFrom, dateTo); filename = `bookings_${dateFrom || 'all'}_${dateTo || 'all'}.csv`; break; case 'clients': csv = await exportClients(supabase, businessId); filename = `clients_${new Date().toISOString().split('T')[0]}.csv`; break; case 'revenue': csv = await exportRevenue(supabase, businessId, dateFrom, dateTo); filename = `revenue_${dateFrom || 'all'}_${dateTo || 'all'}.csv`; break; default: return res(400, 'Invalid export type'); } return { statusCode: 200, headers: { 'Content-Type': 'text/csv; charset=utf-8', 'Content-Disposition': `attachment; filename="${filename}"`, 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Headers': 'Content-Type, Authorization' }, body: csv }; } catch (err) { if (err.status) return res(err.status, err.message); console.error('Export error:', err); return res(500, 'Export failed'); } }; async function exportBookings(supabase, businessId, dateFrom, dateTo) { let query = supabase.from('bookings') .select('id, start_at, end_at, customer_name, customer_phone, customer_email, is_new_client, status, created_at, booking_items(item_type, name_snapshot, price_snapshot, minutes_snapshot)') .eq('business_id', businessId) .order('start_at', { ascending: false }); if (dateFrom) query = query.gte('start_at', dateFrom + 'T00:00:00Z'); if (dateTo) query = query.lte('start_at', dateTo + 'T23:59:59Z'); const { data } = await query.limit(5000); const rows = (data || []).map(b => { const services = (b.booking_items || []).filter(i => i.item_type === 'service').map(i => i.name_snapshot).join('; '); const addons = (b.booking_items || []).filter(i => i.item_type === 'add_on').map(i => i.name_snapshot).join('; '); const prices = (b.booking_items || []).map(i => i.price_snapshot).filter(Boolean).join('; '); const duration = (b.booking_items || []).reduce((sum, i) => sum + (i.minutes_snapshot || 0), 0); return [ b.start_at, b.end_at, b.customer_name, b.customer_phone || '', b.customer_email || '', b.is_new_client ? 'Yes' : 'No', b.status, services, addons, prices, duration, b.created_at ]; }); const header = ['Start', 'End', 'Name', 'Phone', 'Email', 'New Client', 'Status', 'Services', 'Add-Ons', 'Prices', 'Duration (min)', 'Created']; return toCSV(header, rows); } async function exportClients(supabase, businessId) { const { data } = await supabase.from('clients') .select('name, phone, email, total_visits, first_visit_at, last_visit_at, no_show_count, is_no_show_flagged, tags, admin_notes') .eq('business_id', businessId) .order('name') .limit(10000); const rows = (data || []).map(c => [ c.name, c.phone || '', c.email || '', c.total_visits, c.first_visit_at || '', c.last_visit_at || '', c.no_show_count, c.is_no_show_flagged ? 'Yes' : 'No', (c.tags || []).join('; '), c.admin_notes || '' ]); const header = ['Name', 'Phone', 'Email', 'Total Visits', 'First Visit', 'Last Visit', 'No-Shows', 'Flagged', 'Tags', 'Notes']; return toCSV(header, rows); } async function exportRevenue(supabase, businessId, dateFrom, dateTo) { let query = supabase.from('bookings') .select('start_at, status, booking_items(name_snapshot, price_snapshot, item_type)') .eq('business_id', businessId) .in('status', ['confirmed', 'no_show']) .order('start_at', { ascending: false }); if (dateFrom) query = query.gte('start_at', dateFrom + 'T00:00:00Z'); if (dateTo) query = query.lte('start_at', dateTo + 'T23:59:59Z'); const { data } = await query.limit(5000); // Aggregate by date const byDate = {}; for (const b of (data || [])) { const date = b.start_at.split('T')[0]; if (!byDate[date]) byDate[date] = { date, count: 0, minTotal: 0, maxTotal: 0, services: {} }; byDate[date].count++; for (const item of (b.booking_items || [])) { const parsed = parsePrice(item.price_snapshot); if (parsed) { byDate[date].minTotal += parsed.min; byDate[date].maxTotal += parsed.max; } const key = item.name_snapshot; if (!byDate[date].services[key]) byDate[date].services[key] = 0; byDate[date].services[key]++; } } const rows = Object.values(byDate).sort((a, b) => b.date.localeCompare(a.date)).map(d => { const topServices = Object.entries(d.services).sort((a, b) => b[1] - a[1]).slice(0, 3).map(([n, c]) => `${n} (${c})`).join('; '); return [d.date, d.count, '$' + d.minTotal, '$' + d.maxTotal, topServices]; }); const header = ['Date', 'Bookings', 'Revenue (Low)', 'Revenue (High)', 'Top Services']; return toCSV(header, rows); } function parsePrice(str) { if (!str) return null; const nums = str.match(/\d+(\.\d+)?/g); if (!nums?.length) return null; const values = nums.map(Number); return { min: Math.min(...values), max: Math.max(...values) }; } function toCSV(header, rows) { const escape = v => { const s = String(v ?? ''); return s.includes(',') || s.includes('"') || s.includes('\n') ? '"' + s.replace(/"/g, '""') + '"' : s; }; return [header.map(escape).join(','), ...rows.map(r => r.map(escape).join(','))].join('\n'); } function res(code, msg) { return { statusCode: code, headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Headers': 'Content-Type, Authorization' }, body: JSON.stringify({ error: msg }) }; } function cors() { return { statusCode: 204, headers: { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Headers': 'Content-Type, Authorization', 'Access-Control-Allow-Methods': 'GET, OPTIONS' }, body: '' }; }
